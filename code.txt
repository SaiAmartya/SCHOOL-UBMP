#include    "xc.h"              // Microchip XC8 compiler include file
#include    "stdint.h"          // Include integer definitions
#include    "stdbool.h"         // Include Boolean (true/false) definitions

#include    "UBMP420.h"         // Include UBMP4.2 constants and functions
#include <builtins.h>

//Initialize variables here!!!
int loop = 0;
float distance = 0;

// TODO Set linker ROM ranges to 'default,-0-7FF' under "Memory model" pull-down.
// TODO Set linker code offset to '800' under "Additional options" pull-down.

// The main function is required, and the program begins executing from here.

int main(void)
{

    // Configure oscillator and I/O ports. These functions run once at start-up.
    OSC_config();               // Configure internal oscillator for 48 MHz
    UBMP4_config();             // Configure on-board UBMP4 I/O devices
    
    // Code in this while loop runs repeatedly.
    while(1)
	{        
        // send pulse to ultrasonic sensor 
        H5OUT = 0; 
        __delay_us(5);

        H5OUT = 1; 
        __delay_us(10);
        H5OUT = 0; 

        loop = 0;
        
        while (H1IN == 0)
        {
        // Wait for echo to go high    
        }

        while (H1IN == 1)
        {
        // Echo is high, start counting
            __delay_us(10);
            loop++;
        }
        // Calculate distance in cm (divide by 2 because the wave travels there AND back)
        distance = loop  * 0.343 / 2;

        // turn on both motors if distance is less than 30cm
        if (distance > 0 && distance < 30){
            // run for a total of 2 seconds

            // full power 
            H8OUT = 1;
            H6OUT = 1;
            __delay_ms(500);

            // 75% power for 500 ms
            for (int i = 0; i < 5000; i++){
                H8OUT = 1;
                H6OUT = 1;
                __delay_us(75);
                H8OUT = 0;
                H6OUT = 0;
                __delay_us(25);
            }
            
            // 50% power
            for (int i = 0; i < 500; i++){
                H8OUT = 1;
                H6OUT = 1;
                __delay_us(50);
                H8OUT = 0;
                H6OUT = 0;
                __delay_us(50);
            }

            // 25% power
            for (int i = 0; i < 5000; i++){
                H8OUT = 1;
                H6OUT = 1;
                __delay_us(25);
                H8OUT = 0;
                H6OUT = 0;
                __delay_us(75);
            }
        }
        else {
            H8OUT = 0;
            H6OUT = 0;
        }

        // Activate bootloader if SW1 is pressed.
        if(SW1 == 0)
        {
            RESET();
        }
    }
}
